{% highlight js %} import Turbolinks from 'turbolinks' const { beyond } = window Turbolinks.start() document.addEventListener('turbolinks:before-render', () => beyond.unbindAll()) document.addEventListener('turbolinks:render', () => beyond.bind()) beyond.docReady() .then(() => beyond.bind()) {% endhighlight %}
